247  bool configure_(const std::string& stropt, fas::true_, json::json_error* err)
248  {
249  component_config opt;
250  if ( !this->unserialize_(opt, stropt, err ) )
251  return false;
252  opt.name = this->name();
253 
254  config_pair op;
255  op.basic = this->serialize_basic_(opt);
256  op.custom = this->serialize_custom_(opt);
257 
258  if ( _config.basic.empty() )
259  {
260  // Это дефолтная конфигурация синглетона при создании объекта
261  _instances->configure(opt);
262  }
263  else if ( _config.custom != op.custom )
264  {
265  _instances->reconfigure(opt);
266  SYSTEM_LOG_MESSAGE("Singleton '" << opt.name << "' is reconfigured (fully)")
267  }
268  else if ( _config.basic != op.basic )
269  {
270  _instances->reconfigure_basic(opt);
271  SYSTEM_LOG_MESSAGE("Singleton '" << opt.name << "' is reconfigured (basic)")
272  }
273  else
274  {
275  SYSTEM_LOG_MESSAGE("Singleton '" << opt.name << "' without reconfiguration")
276  }
277  _config = op;
278  return true;
279  }

292  bool configure_(const std::string& stropt, fas::false_, json::json_error* err)
293  {
294  std::set<std::string> stop_list;
295  for (const auto& item : _instances )
296  stop_list.insert(item.first);
297 
298  component_config optlist;
299  if ( !this->unserialize_( optlist, stropt, err ) )
300  return false;
301 
302  for (const auto& opt : optlist )
303  {
304  stop_list.erase(opt.name);
305  config_pair op;
306  op.basic = this->serialize_basic_(opt);
307  op.custom = this->serialize_custom_(opt);
308 
309  auto itr = _instances.find(opt.name);
310  if ( itr == _instances.end() )
311  {
312  auto inst = std::make_shared<instance_type>();
313  itr = _instances.insert( std::make_pair( opt.name, inst) ).first;
314  if ( _global ) _global->registry.set_object("instance", opt.name, inst);
315  inst->create( _global );
316  itr->second->configure(opt);
317  SYSTEM_LOG_MESSAGE("Instance '" << opt.name << "' is initial configured")
318  _config.insert(std::make_pair(opt.name, op) );
319  }
320  else
321  {
322  auto oitr = _config.find(opt.name);
323  if ( oitr == _config.end() ) abort();
324 
325  if ( oitr->second.custom != op.custom )
326  {
327  reconfigure_fully_(itr->second, opt, fas::bool_<ignore_reconfigure!=0>() );
328  }
329  else if ( oitr->second.basic != op.basic )
330  {
331  itr->second->reconfigure_basic(opt);
332  SYSTEM_LOG_MESSAGE("Instance '" << opt.name << "' is reconfigured (basic)")
333  }
334  else
335  {
336  SYSTEM_LOG_MESSAGE("Instance '" << opt.name << "' without reconfiguration")
337  }
338  oitr->second = op;
339  }
340  } // for optlist
341 
342  for ( const auto& n: stop_list )
343  {
344  auto itr = _instances.find(n);
345  if ( itr == _instances.end() )
346  {
347  if ( _global ) _global->registry.erase("instance", n);
348  itr->second->stop("");
349  _instances.erase(itr);
350  SYSTEM_LOG_MESSAGE("Instance '" << n << "' is deleted")
351  }
352  }
353  return true;
354  }
