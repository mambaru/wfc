79  virtual void perform_incoming(incoming_holder holder, io_id_t io_id, outgoing_handler_t handler) override
80  {
81  if ( handler == nullptr )
82  {
83  handler = [](outgoing_holder oholder)
84  {
85  JSONRPC_LOG_WARNING( "target_adapter::perform_incoming stub:" << oholder.detach() )
86  };
87  }
88 
89  if ( auto p1 = _jsonrpc.lock() )
90  {
91  p1->perform_incoming( std::move(holder), io_id, handler);
92  }
93  else if ( auto p2 = _itf.lock() )
94  {
95  if ( handler!=nullptr )
96  {
97  this->perform_io_( holder.detach(), io_id, [handler](data_ptr d)
98  {
99  handler( outgoing_holder( std::move(d) ) );
100  });
101  }
102  else
103  this->perform_io_( holder.detach(), io_id, [handler](data_ptr d){
104  JSONRPC_LOG_WARNING( "target_adapter::perform_incoming->perform_io_ stub:" << d )
105  } );
106  }
107  else if ( handler != nullptr )
108  {
109  ::wjrpc::aux::send_error(std::move(holder), std::make_unique< ::wjrpc::service_unavailable > (), std::move(handler));
110  }
111  else
112  {
113  }
114  }
115 
116  virtual void perform_outgoing(outgoing_holder holder, io_id_t io_id) override
117  {
118  if ( auto p1 = _jsonrpc.lock() )
119  {
120  p1->perform_outgoing( std::move(holder), io_id);
121  }
122  else if ( auto p2 = _itf.lock() )
123  {
124  auto handler = holder.result_handler();
125  auto d = holder.detach();
126 
127  p2->perform_io( std::move(d), io_id, [handler](data_ptr d2)
128  {
129  while ( d2!=nullptr )
130  {
131  incoming_holder iholder( std::move(d2) );
132  wjson::json_error e;
133  d2 = iholder.parse(&e);
134  if ( !e && iholder )
135  {
136  handler( std::move(iholder) );
137  }
138  else
139  {
140  ijsonrpc::outgoing_handler_t err_handler = [handler](outgoing_holder err)
141  {
142  auto d3 = err.detach();
143  handler( incoming_holder( std::move(d3)));
144  };
145 
146  wjrpc::aux::send_error(std::move(iholder), std::make_unique< wjrpc::parse_error>(), err_handler );
147  }
148  }
149  });
150  }
151  else if ( auto handler = holder.result_handler() )
152  {
153  if ( auto call_id = holder.call_id() )
154  {
155  auto eholder = create_error_holder_< wjrpc::service_unavailable >(call_id);
156  handler( std::move(eholder) );
157  }
158  else
159  {
160  handler( incoming_holder( data_ptr() ) );
161  }
162  }
163  }
164 private:
165  void perform_io_(data_ptr d, io_id_t io_id, output_handler_t handler)
166  {
167  if ( auto p = _itf.lock() )
168  {
169  p->perform_io( std::move(d), io_id, std::move(handler) );
170  }
171  else if ( handler != nullptr)
172  {
173  handler(nullptr);
174  }
175  }
176 
177  template<typename Error>
178  incoming_holder create_error_holder_(call_id_t call_id)
179  {
180  wjrpc::outgoing_error< wjrpc::error > err;
181  err.error = std::make_unique<Error>();
182  err.id = std::make_unique<data_type>();
183  wjson::value<call_id_t>::serializer()( call_id, std::back_inserter(*(err.id)) );
184 
185  auto d = std::make_unique<data_type>();
186  d->reserve(100);
187  wjrpc::outgoing_error_json< wjrpc::error_json >::serializer()(err, std::back_inserter(*d));
188  incoming_holder holder( std::move(d) );
189  holder.parse(nullptr);
190  return holder;
191  }
192 private:
193 
194  itf_ptr_t _itf;
195  jsonrpc_ptr_t _jsonrpc;
196 };
