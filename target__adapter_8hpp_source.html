23 struct target_adapter: ijsonrpc
24 {
25  typedef std::weak_ptr<iinterface> itf_ptr_t;
26  typedef std::weak_ptr<ijsonrpc> jsonrpc_ptr_t;
27  typedef ::wjrpc::call_id_t call_id_t;

29  target_adapter() = default;
30  target_adapter(itf_ptr_t itf, jsonrpc_ptr_t jsonrpc)
31  : _itf(itf), _jsonrpc(jsonrpc)
32  {}
33 
34  operator bool () const
35  {
36  return _itf.lock()!=nullptr || _jsonrpc.lock()!=nullptr;
37  }

39  virtual void reg_io(io_id_t io_id, std::weak_ptr<iinterface> itf) override
40  {

42  {
43  p->reg_io(io_id, itf);
44  }
45  }
46  virtual void unreg_io(io_id_t io_id) override
47  {
48  if ( auto p = _itf.lock() )
49  {
50  p->unreg_io(io_id);
51  }
52  }
53 
54  virtual void perform_io(data_ptr d, io_id_t io_id, output_handler_t handler) override
55  {
56  if ( auto p = _itf.lock() )
57  {
58  p->perform_io( std::move(d), io_id, std::move(handler) );
59  }
60  else if ( handler != nullptr)
61  {
62  handler(nullptr);
63  }
64  }
65 
66  virtual void perform_incoming(incoming_holder holder, io_id_t io_id, outgoing_handler_t handler) override
67  {
68  if ( auto p1 = _jsonrpc.lock() )
69  {
70  p1->perform_incoming( std::move(holder), io_id, handler);
71  }
72  else if ( auto p2 = _itf.lock() )
73  {
74  if ( handler!=nullptr )
75  {
76  this->perform_io( holder.detach(), io_id, [handler](data_ptr d)
77  {
78  //outgoing_holder holder;
79  handler( outgoing_holder( std::move(d) ) );
80  });
81  }
82  else
83  this->perform_io( holder.detach(), io_id, nullptr );
84  }
85  else if ( handler != nullptr )
86  {
87  ::wjrpc::aux::send_error(std::move(holder), std::make_unique< ::wjrpc::service_unavailable > (), std::move(handler));
88  }
89  else
90  {
91  }
92  }
93 
94  template<typename Error>
95  incoming_holder create_error_holder_(call_id_t call_id)
96  {
97  wjrpc::outgoing_error< wjrpc::error > err;
98  err.error = std::make_unique<Error>();
99  err.id = std::make_unique<data_type>();
100  wjson::value<call_id_t>::serializer()( call_id, std::back_inserter(*(err.id)) );
101 
102  auto d = std::make_unique<data_type>();
103  d->reserve(100);
104  wjrpc::outgoing_error_json< wjrpc::error_json >::serializer()(err, std::back_inserter(*d));
105  incoming_holder holder( std::move(d) );
106  holder.parse(nullptr);
107  return holder;
108  }
109 
110  virtual void perform_outgoing(outgoing_holder holder, io_id_t io_id) override
111  {
112  if ( auto p1 = _jsonrpc.lock() )
113  {
114  p1->perform_outgoing( std::move(holder), io_id);
115  }
116  else if ( auto p2 = _itf.lock() )
117  {
118  auto handler = holder.result_handler();
119  auto d = holder.detach();
120 
121  p2->perform_io( std::move(d), io_id, [handler](data_ptr d2)
122  {
123  while ( d2!=nullptr )
124  {
125  incoming_holder iholder( std::move(d2) );
126  wjson::json_error e;
127  d2 = iholder.parse(&e);
128  if ( !e && iholder )
129  {
130  handler( std::move(iholder) );
131  }
132  else
133  {
134  ijsonrpc::outgoing_handler_t err_handler = [handler](outgoing_holder err)
135  {
136  auto d3 = err.detach();
137  handler( incoming_holder( std::move(d3)));
138  };
139 
140  wjrpc::aux::send_error(std::move(iholder), std::make_unique< wjrpc::parse_error>(), err_handler );
141  }
142  }
143  });
144  }
145  else if ( auto handler = holder.result_handler() )
146  {
147  if ( auto call_id = holder.call_id() )
148  {
149  auto eholder = create_error_holder_< wjrpc::service_unavailable >(call_id);
150  handler( std::move(eholder) );
151  }
152  else
153  {
154  handler( incoming_holder( data_ptr() ) );
155  }
156  }
157  }

161  itf_ptr_t _itf;
162  jsonrpc_ptr_t _jsonrpc;
163 };
