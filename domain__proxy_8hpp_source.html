55  virtual void perform_io(data_ptr d, io_id_t io_id, output_handler_t handler) override
56  {
57  if ( this->suspended() )
58  {
59  _adapter.perform_io( std::move(d), io_id, handler );
60  return;
61  }
62 
63  while ( d != nullptr )
64  {
65  incoming_holder holder( std::move(d));
66 
67  wjson::json_error e;
68  d = holder.parse(&e);
69  if ( holder.is_valid() )
70  {
71  this->perform_incoming( std::move(holder), io_id, [handler](outgoing_holder oholder)
72  {
73  handler( oholder.detach() );
74  });
75  }
76  else
77  {
78  outgoing_handler_t error_handler = nullptr;
79  if ( holder.has_id() || e )
80  {
81  error_handler = [handler](outgoing_holder oholder)
82  {
83  auto d2 = oholder.detach();
84  if ( d2!=nullptr ) { JSONRPC_LOG_ERROR( d2 )}
85  handler( std::move(d2) );
86  };
87  }
88 
89  if ( e )
90  {
91  JSONRPC_LOG_ERROR( "domain_proxy: Parse error: " << holder.error_message(e)
92  << " error_handler: "<< (error_handler != nullptr) );
93  this->send_error< wjrpc::parse_error >( std::move(holder), error_handler);
94  }
95  else
96  {
97  JSONRPC_LOG_ERROR( "domain_proxy: Invalid JSON-RPC: " << holder.str() );
98  this->send_error< wjrpc::invalid_request >( std::move(holder), error_handler);
99  }
100  }
101  }
102  }
