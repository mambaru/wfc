53  virtual void perform_io(data_ptr d, io_id_t io_id, output_handler_t handler) override
54  {
55  if ( this->suspended() )
56  {
57  _adapter.perform_io( std::move(d), io_id, handler );
58  return;
59  }
60 
61  while ( d != nullptr )
62  {
63  incoming_holder holder( std::move(d));
64 
65  wjson::json_error e;
66  d = holder.parse(&e);
67  if ( holder.is_valid() )
68  {
69  outgoing_handler_t jsonrpc_handler = nullptr;
70  if ( holder.is_request() )
71  jsonrpc_handler = [handler](outgoing_holder holder2) { handler( holder2.detach() ); };
72  this->perform_incoming( std::move(holder), io_id, jsonrpc_handler);
73  }
74  else
75  {
76  outgoing_handler_t error_handler = nullptr;
77  if ( holder.has_id() )
78  {
79  error_handler = [handler](outgoing_holder oholder)
80  {
81  auto d2 = oholder.detach();
82  if ( d2!=nullptr ) { JSONRPC_LOG_ERROR( d2 )}
83  handler( std::move(d2) );
84  };
85  }
86 
87  if ( e )
88  {
89  JSONRPC_LOG_ERROR( "domain_proxy: Parse error: " << holder.error_message(e) );
90  this->send_error< wjrpc::parse_error >( std::move(holder), error_handler);
91  }
92  else
93  {
94  JSONRPC_LOG_ERROR( "domain_proxy: Invalid JSON-RPC: " << holder.str() );
95  this->send_error< wjrpc::invalid_request >( std::move(holder), error_handler);
96  }
97  }
98  }
99  }
