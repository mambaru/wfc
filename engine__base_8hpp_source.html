119  static std::string schema_help(const std::string& component_name, const std::string& args, const std::string& base_help )
120  {
121  typedef typename engine_type::handler_type handler_type;
122  std::stringstream ss;
123  schema_map_t schema_map;
124  std::list<std::string> method_list = get_method_list(schema_map);
125  if ( args.empty() )
126  {
127  ss << base_help << std::endl;
128  std::string method_ws_schema;
129  ss << "Allowed methods for '" << component_name << "' [allowed schemas]:" << std::endl;
130  for (auto n : method_list)
131  {
132  ss << " " << n;
133  auto method_schemas = schema_map[n];
134  if ( !method_schemas.empty() )
135  {
136  std::stringstream ss1;
137  bool has_schema_help = !method_ws_schema.empty();
138  if ( !has_schema_help )
139  {
140  ss1 << "\t--help " << component_name << ":schema:" << n << "=";
141  }
142 
143  ss << " [";
144  for (size_t i = 0; i < method_schemas.size() ; ++i)
145  {
146  ss << method_schemas[i];
147  ss1 << method_schemas[i];
148  if ( i == method_schemas.size() - 1)
149  {
150  ss << "]";
151  }
152  else
153  {
154  ss << ",";
155  ss1 << ",";
156  }
157  }
158  method_ws_schema = ss1.str();
159  }
160  ss << std::endl;
161  }
162 
163  if ( !method_list.empty() )
164  {
165  ss << "Get the JSON-RPC default schema:"<< std::endl;
166  ss << "\t--help " << component_name << ":schema" << std::endl;
167  ss << "\tExamples for method "<< method_list.front() << ":" << std::endl;
168  ss << "\t--help " << component_name << ":schema:" << method_list.front() << std::endl;
169  ss << "\t--help " << component_name << ":params:" << method_list.front() << std::endl;
170  ss << "\t--help " << component_name << ":result:" << method_list.front() << std::endl;
171  ss << "\t--help " << component_name << ":request:" << method_list.front() << std::endl;
172  ss << "\t--help " << component_name << ":response:" << method_list.front();
173 
174  if ( !method_ws_schema.empty() )
175  {
176  ss << std::endl;
177  ss << "All allowed schemas: " << method_ws_schema << std::endl;
178  ss << "Get the JSON-RPC named schema example:"<< std::endl;
179  ss << "\t--help " << component_name << ":schema:" << method_list.front() << "=" << method_ws_schema;
180  }
181  }
182  }
183  else
184  {
185  typedef typename handler_type::schema_list_t schema_list_t;
186  typedef typename schema_list_t::value_type schema_type;
187  typedef typename schema_type::json schema_json;
188  typedef typename schema_json::list schema_list_json;
189 
190  size_t eqpos=args.find('=');
191  std::string argname = args.substr(0, eqpos);
192  std::string argval = ((eqpos!=std::string::npos) ? args.substr(eqpos+1) : std::string());
193  std::vector<std::string> schema_list;
194  boost::split( schema_list, argval, boost::is_any_of(",") );
195  size_t schpos = argname.find(':');
196  std::string schema_tname = argname.substr(0, schpos);
197  std::string method_name = schpos!=std::string::npos ? argname.substr(schpos+1) : std::string();
198 
199  schema_list_t schema = handler_type::create_schema(schema_list);
200  if ( schema_tname=="schema" && method_name.empty() )
201  {
202  typename schema_list_json::serializer()(schema, std::ostreambuf_iterator<char>(ss) );
203  }
204  else
205  {
206  auto itr = std::find_if(schema.begin(), schema.end(),
207  [method_name](const schema_type& sch)->bool{
208  return sch.name == method_name;
209  }
210  );
211 
212  if ( itr!=schema.end() )
213  {
214  using namespace wjson::literals;
215  if ( schema_tname=="schema" )
216  typename schema_json::serializer()(*itr, std::ostreambuf_iterator<char>(ss) );
217  else if ( schema_tname=="params" )
218  ss << itr->params;
219  else if ( schema_tname=="result" )
220  ss << itr->result;
221  else if ( schema_tname=="request" )
222  ss << "{'jsonrpc':'2.0','method':'"_json << method_name << "','params':"_json << itr->params << ",'id':1}"_json;
223  else if ( schema_tname=="response" )
224  ss << "{'jsonrpc':'2.0','result':"_json << itr->result << ",'id':1}"_json;
225  else
226  ss << "Error: Invalid argument '" << schema_tname << "'";
227  }
228  else
229  ss << "ERROR: procedure '"<< method_name<<"' not found";
230  }
231  }
232  return ss.str();
233  }

251  void initialize_engine(options_type eopt)
252  {
253  eopt.domain_name = this->name();
254  eopt.powner = &(this->owner());
255  eopt.tracking_id = this->get_id();
256  eopt.disable_tracking_log = false;
257 
258  if ( _engine == nullptr )
259  {
260  _engine = std::make_shared<engine_type>();
261  _engine->start(eopt, ::iow::io::create_id<size_t>() );
262  }
263  else
264  {
265  _engine->reconfigure( eopt );
266  }
267 
268  if ( auto wf = this->get_workflow() )
269  {
270  wf->release_timer( _timer_id );
271  if ( eopt.remove_outdated_ms != 0 )
272  {
273  auto pengine = this->_engine;
274  _timer_id = wf->create_timer(
275  std::chrono::milliseconds(eopt.remove_outdated_ms),
276  [pengine]() -> bool
277  {
278  JSONRPC_LOG_DEBUG( " JSON-RPC remove outdated...");
279  if ( size_t count = pengine->remove_outdated() )
280  {
281  JSONRPC_LOG_WARNING( "JSON-RPC: " << count << " calls is outdated.");
282  }
283  return true;
284  }
285  );
286  }
287  }
288  this->initialize_statistics_();
289  }

291 private:
292  void initialize_statistics_()
293  {
294  if ( auto stat = this->get_statistics() )
295  {
296  auto sopt = this->statistics_options();
297  if ( auto wf = this->get_workflow() )
298  {
299  wf->release_timer(_stat_timer_id);
300  if ( sopt.interval_ms != 0 )
301  {
302  auto sname = this->name();
303  auto handler_map = stat->create_value_meter(sname + sopt.handler_map);
304  auto result_map = stat->create_value_meter(sname + sopt.result_map);
305  auto result_queue = stat->create_value_meter(sname + sopt.result_queue);
306  _stat_timer_id = wf->create_timer(
307  std::chrono::milliseconds(sopt.interval_ms),
308  [handler_map, result_map, result_queue, this]()->bool
309  {
310  if ( nullptr != this->get_statistics() )
311  {
312  auto si = this->_engine->sizes();
313  handler_map.create( static_cast<wrtstat::value_type>(si.handler_map), static_cast<wrtstat::size_type>(0));
314  result_map.create( static_cast<wrtstat::value_type>(si.result_map), static_cast<wrtstat::size_type>(0));
315  result_queue.create(static_cast<wrtstat::value_type>(si.result_queue), static_cast<wrtstat::size_type>(0));
316  return true;
317  }
318  return false;
319  }
320  );
321  }
322  }
323  }
