120  static std::string schema_help(const std::string& component_name, const std::string& args, const std::string& base_help )
121  {
122  typedef typename engine_type::handler_type handler_type;
123  std::stringstream ss;
124  schema_map_t schema_map;
125  std::list<std::string> method_list = get_method_list(schema_map);
126  if ( args.empty() )
127  {
128  ss << base_help << std::endl;
129  std::string method_ws_schema;
130  ss << "Allowed methods for '" << component_name << "' [allowed schemas]:" << std::endl;
131  for (auto n : method_list)
132  {
133  ss << " " << n;
134  auto method_schemas = schema_map[n];
135  if ( !method_schemas.empty() )
136  {
137  std::stringstream ss1;
138  bool has_schema_help = !method_ws_schema.empty();
139  if ( !has_schema_help )
140  {
141  ss1 << "\t--help " << component_name << ":schema:" << n << "=";
142  }
143 
144  ss << " [";
145  for (size_t i = 0; i < method_schemas.size() ; ++i)
146  {
147  ss << method_schemas[i];
148  ss1 << method_schemas[i];
149  if ( i == method_schemas.size() - 1)
150  {
151  ss << "]";
152  }
153  else
154  {
155  ss << ",";
156  ss1 << ",";
157  }
158  }
159  method_ws_schema = ss1.str();
160  }
161  ss << std::endl;
162  }
163 
164  if ( !method_list.empty() )
165  {
166  ss << "Get the JSON-RPC default schema:"<< std::endl;
167  ss << "\t--help " << component_name << ":schema" << std::endl;
168  ss << "\tExamples for method "<< method_list.front() << ":" << std::endl;
169  ss << "\t--help " << component_name << ":schema:" << method_list.front() << std::endl;
170  ss << "\t--help " << component_name << ":params:" << method_list.front() << std::endl;
171  ss << "\t--help " << component_name << ":result:" << method_list.front() << std::endl;
172  ss << "\t--help " << component_name << ":request:" << method_list.front() << std::endl;
173  ss << "\t--help " << component_name << ":response:" << method_list.front();
174 
175  if ( !method_ws_schema.empty() )
176  {
177  ss << std::endl;
178  ss << "All allowed schemas: " << method_ws_schema << std::endl;
179  ss << "Get the JSON-RPC named schema example:"<< std::endl;
180  ss << "\t--help " << component_name << ":schema:" << method_list.front() << "=" << method_ws_schema;
181  }
182  }
183  }
184  else
185  {
186  typedef typename handler_type::schema_list_t schema_list_t;
187  typedef typename schema_list_t::value_type schema_type;
188  typedef typename schema_type::json schema_json;
189  typedef typename schema_json::list schema_list_json;
190 
191  size_t eqpos=args.find('=');
192  std::string argname = args.substr(0, eqpos);
193  std::string argval = ((eqpos!=std::string::npos) ? args.substr(eqpos+1) : std::string());
194  std::vector<std::string> schema_list;
195  boost::split( schema_list, argval, boost::is_any_of(",") );
196  size_t schpos = argname.find(':');
197  std::string schema_tname = argname.substr(0, schpos);
198  std::string method_name = schpos!=std::string::npos ? argname.substr(schpos+1) : std::string();
199 
200  schema_list_t schema = handler_type::create_schema(schema_list);
201  if ( schema_tname=="schema" && method_name.empty() )
202  {
203  typename schema_list_json::serializer()(schema, std::ostreambuf_iterator<char>(ss) );
204  }
205  else
206  {
207  auto itr = std::find_if(schema.begin(), schema.end(),
208  [method_name](const schema_type& sch)->bool{
209  return sch.name == method_name;
210  }
211  );
212 
213  if ( itr!=schema.end() )
214  {
215  using namespace wjson::literals;
216  if ( schema_tname=="schema" )
217  typename schema_json::serializer()(*itr, std::ostreambuf_iterator<char>(ss) );
218  else if ( schema_tname=="params" )
219  ss << itr->params;
220  else if ( schema_tname=="result" )
221  ss << itr->result;
222  else if ( schema_tname=="request" )
223  ss << "{'jsonrpc':'2.0','method':'"_json << method_name << "','params':"_json << itr->params << ",'id':1}"_json;
224  else if ( schema_tname=="response" )
225  ss << "{'jsonrpc':'2.0','result':"_json << itr->result << ",'id':1}"_json;
226  else
227  ss << "Error: Invalid argument '" << schema_tname << "'";
228  }
229  else
230  ss << "ERROR: procedure '"<< method_name<<"' not found";
231  }
232  }
233  return ss.str();
234  }

252  void initialize_engine(options_type eopt)
253  {
254  if ( _engine == nullptr )
255  {
256  _engine = std::make_shared<engine_type>();
257  _engine->start(eopt, ::iow::io::create_id<size_t>() );

264  if ( auto wf = this->get_workflow() )
265  {
266  wf->release_timer( _timer_id );
267  if ( eopt.remove_outdated_ms != 0 )
268  {
269  auto pengine = this->_engine;
270  _timer_id = wf->create_timer(
271  std::chrono::milliseconds(eopt.remove_outdated_ms),
272  [pengine]() -> bool
273  {
274  JSONRPC_LOG_DEBUG( " JSON-RPC remove outdated...");
275  if ( size_t count = pengine->remove_outdated() )
276  {
277  JSONRPC_LOG_WARNING( "JSON-RPC: " << count << " calls is outdated.");
278  }
279  return true;
280  }
281  );
282  }
283  }
284  this->initialize_statistics_();
285  }

288  void initialize_statistics_()
289  {
290  if ( auto stat = this->get_statistics() )
291  {
292  auto sopt = this->statistics_options();
293  if ( auto wf = this->get_workflow() )
294  {
295  wf->release_timer(_stat_timer_id);
296  if ( sopt.interval_ms != 0 )
297  {
298  auto sname = this->name();
299  auto handler_map = stat->create_value_meter(sname + sopt.handler_map);
300  auto result_map = stat->create_value_meter(sname + sopt.result_map);
301  auto result_queue = stat->create_value_meter(sname + sopt.result_queue);
302  _stat_timer_id = wf->create_timer(
303  std::chrono::milliseconds(sopt.interval_ms),
304  [handler_map, result_map, result_queue, this]()->bool
305  {
306  if ( nullptr != this->get_statistics() )
307  {
308  auto si = this->_engine->sizes();
309  handler_map.create( static_cast<wrtstat::value_type>(si.handler_map), static_cast<wrtstat::size_type>(0));
310  result_map.create( static_cast<wrtstat::value_type>(si.result_map), static_cast<wrtstat::size_type>(0));
311  result_queue.create(static_cast<wrtstat::value_type>(si.result_queue), static_cast<wrtstat::size_type>(0));
312  return true;
313  }
314  return false;
315  }
316  );
317  }
318  }
319  }
