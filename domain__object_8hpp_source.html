927  /* 0-строка, 1=опции */
928  int dual_target_deserialize_(const std::string& target, std::string* str, wflow::workflow_options* opt) const
929  {
930  if ( target.empty() )
931  return 0;
932 
933  wjson::json_error er;
934  int ret = wjson::multi_deserializer<
935  wjson::string<>,
936  wflow::workflow_options_json
937  >( target.begin(), target.end(), *str, *opt, &er );
938 
939  if ( ret > 1 )
940  {
941  ret = 0;
942 
943  if ( er )
944  {
945  *str = target;
946  er.reset();
947  }
948  else if ( !er )
949  {
950  wflow::workflow_options tmp;
951  wflow::workflow_options_json::serializer()(tmp, target.begin(), target.end(), &er);
952  str->clear();
953  }
954 
955  if ( er )
956  {
957  SYSTEM_LOG_ERROR("Workflow configuration error. Default configuration will be used. Trace:"
958  << wjson::strerror::message_trace(er, target.begin(), target.end()));
959  }
960  }
961  return ret;
962  }
