933  /* 0-строка, 1=опции */
934  int dual_target_deserialize_(const std::string& target, std::string* str, wflow::workflow_options* opt) const
935  {
936  if ( target.empty() )
937  return 0;
938 
939  wjson::json_error er;
940  int ret = wjson::multi_deserializer<
941  wjson::string<>,
942  wflow::workflow_options_json
943  >( target.begin(), target.end(), *str, *opt, &er );
944 
945  if ( ret > 1 )
946  {
947  ret = 0;
948 
949  if ( er )
950  {
951  *str = target;
952  er.reset();
953  }
954  else if ( !er )
955  {
956  wflow::workflow_options tmp;
957  wflow::workflow_options_json::serializer()(tmp, target.begin(), target.end(), &er);
958  str->clear();
959  }
960 
961  if ( er )
962  {
963  SYSTEM_LOG_ERROR("Workflow configuration error. Default configuration will be used. Trace:"
964  << wjson::strerror::message_trace(er, target.begin(), target.end()));
965  }
966  }
967  return ret;
968  }
