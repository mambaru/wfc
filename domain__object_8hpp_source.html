315  bool perform_status(data_ptr& d, output_handler_t& handler)
316  {
317  if ( handler == nullptr )
318  return false;
319 
320  if ( d == nullptr )
321  {
322  handler(nullptr);
323  return true;
324  }
325 
326  if ( this->suspended() )
327  {
328  handler( iow::io::make("SUSPENDED") );
329  return true;
330  }
331 
332  if ( d->size() < 7 )
333  return false;
334 
335  std::istringstream iss( std::string( d->begin(), d->end()) );
336 
337  std::string command;
338  iss >> command;
339 
340  if ( command != "@STATUS" )
341  return false;
342 
343  size_t errors = 0, warnings = 0;
344 
345  if ( iss >> errors )
346  iss >> warnings;
347 
348  // Модуль ядра всегда есть, отключаем abort для тестирования
349  if ( auto pcore = this->get_target<icore>("core", true) )
350  {
351  std::string statustext = pcore->get_status_text(errors, warnings);
352  handler( iow::io::make(statustext) );
353  }
354  else
355  handler( iow::io::make("OK") );
356  return true;
357  }

996  /* 0-строка, 1=опции */
997  int dual_target_deserialize_(const std::string& target, std::string* str, wflow::workflow_options* opt) const
998  {
999  if ( target.empty() )
1000  return 0;
1001 
1002  wjson::json_error er;
1003  int ret = wjson::multi_deserializer<
1004  wjson::string<>,
1005  wflow::workflow_options_json
1006  >( target.begin(), target.end(), *str, *opt, &er );
1007 
1008  if ( ret > 1 )
1009  {
1010  ret = 0;
1011 
1012  if ( er )
1013  {
1014  *str = target;
1015  er.reset();
1016  }
1017  else if ( !er )
1018  {
1019  wflow::workflow_options tmp;
1020  wflow::workflow_options_json::serializer()(tmp, target.begin(), target.end(), &er);
1021  str->clear();
1022  }
1023 
1024  if ( er )
1025  {
1026  SYSTEM_LOG_ERROR("Workflow configuration error. Default configuration will be used. Trace:"
1027  << wjson::strerror::message_trace(er, target.begin(), target.end()));
1028  }
1029  }
1030  return ret;
1031  }

1053  std::shared_ptr<workflow_type> get_workflow_(const std::string& target, bool disabort, bool* created) const
1054  {
1055  if ( created!=nullptr )
1056  *created = false;
1057 
1058  if ( target.empty() )
1059  return nullptr;
1060 
1061  auto beg = target.begin();
1062  auto end = target.end();
1063 
1064  if ( wjson::parser::is_null(beg, end) )
1065  return nullptr;
1066 
1067  if ( wjson::parser::is_object(beg, end) )
1068  {
1069  if ( created!=nullptr ) *created = true;
1070  return this->create_workflow("", target);
1071  }
1072 
1073  std::string starget;
1074 
1075  if ( wjson::parser::is_string(beg, end) )
1076  wjson::string<>::serializer()(starget, beg, end, nullptr);
1077  else
1078  starget=target;
1079 
1080  if ( starget.empty() )
1081  return nullptr;
1082 
1083  return this->global()->registry.template get_object<workflow_type>("workflow", starget, disabort);
1084  }
