961  int dual_target_deserialize_(const std::string& target, std::string* str, wflow::workflow_options* opt) const
962  {
963  if ( target.empty() )
964  return 0;
965 
966  wjson::json_error er;
967  int ret = wjson::multi_deserializer<
968  wjson::string<>,
969  wflow::workflow_options_json
970  >( target.begin(), target.end(), *str, *opt, &er );
971 
972  if ( ret > 1 )
973  {
974  ret = 0;
975 
976  if ( er )
977  {
978  *str = target;
979  er.reset();
980  }
981  else if ( !er )
982  {
983  wflow::workflow_options tmp;
984  wflow::workflow_options_json::serializer()(tmp, target.begin(), target.end(), &er);
985  str->clear();
986  }
987 
988  if ( er )
989  {
990  SYSTEM_LOG_ERROR("Workflow configuration error. Default configuration will be used. Trace:"
991  << wjson::strerror::message_trace(er, target.begin(), target.end()));
992  }
993  }
994  return ret;
995  }
