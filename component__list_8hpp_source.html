44  virtual std::vector< std::shared_ptr<icomponent> > components() const override
45  {
46  std::vector< std::shared_ptr<icomponent> > result;
47  for (auto& p : _components)
48  {
49  result.push_back(p );
50  }
51  return result;
52  }
53 
54  // only for external control
55  virtual void start(const std::string& arg) override
56  {

60  }
61  }
62 
63  virtual void stop(const std::string& arg) override
64  {
65  for (auto& p : _components)

67  p->stop(arg);
68  }
69  }

71 private:

75  template<typename H, typename L>
76  void create_( fas::type_list< H, L > )
77  {
78  auto obj = std::make_shared<H>();
79  //_components[ obj->name() ] = obj;

81  if ( _global )
82  {

84  }

87  }

89 private:
