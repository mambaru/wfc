20 template<typename ...Args>
21 class fire_list< std::function<bool(Args...)> >
22 {
23  typedef std::recursive_mutex mutex_type;
24 public:
25  typedef std::function<bool(Args...)> fire_fun;
26  typedef std::map< size_t, fire_fun > list_type;
27  typedef size_t handler_id_t;
28 
29  handler_id_t insert( fire_fun f)
30  {
31  std::lock_guard<mutex_type> lk(_mutex);
32  _fire_list.insert( std::make_pair(_counter, f));
33  return _counter++;
34  }
35 
36  void erase(handler_id_t id)
37  {
38  std::lock_guard<mutex_type> lk(_mutex);
39  _fire_list.erase(id);
40  }
41 
42  void clear()
43  {
44  std::lock_guard<mutex_type> lk(_mutex);
45  _fire_list.clear();
46  }
47 
48  void size() const
49  {
50  std::lock_guard<mutex_type> lk(_mutex);
51  return _fire_list.size();
52  }
53 
54  void fire(Args... args) const
55  {
56  list_type lst;
57  {
58  std::lock_guard<mutex_type> lk(_mutex);
59  lst = _fire_list;
60  }
61 
62  auto beg = lst.begin();
63  auto end = lst.end();
64  for (;beg!=end;)
65  {
66  if ( beg->second(args...) )
67  {
68  ++beg;
69  }
70  else
71  {
72  lst.erase(beg++);
73  }
74  }
75  }
76 
77 private:
78  size_t _counter = 1;
79  list_type _fire_list;
80  mutable mutex_type _mutex;
81 };
